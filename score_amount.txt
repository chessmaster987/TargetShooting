#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    double x,y;
    cout<<"x = "; cin>>x; 
    cout<<"y = "; cin>>y;

    const int score[] = {10, 8, 6, 4, 2, 1, 0};
    int r = ceil(sqrt(x*x+y*y));
    int index = (r < 6)?r:6;

    cout<<"score ="<<score[index]<<endl;
    return 0;
}
